document.addEventListener("DOMContentLoaded", function () { const form = document.getElementById("registration-form"); const errorMessage = document.getElementById("error-message"); const usernameInput = document.getElementById("username"); const passwordInput = document.getElementById("password"); const confirmPasswordInput = document.getElementById("confirm-password"); form.addEventListener("submit", function (e) { e.preventDefault(); if (validateForm()) { form.submit(); } }); function validateForm() { let isValid = true; errorMessage.textContent = ""; if (!usernameInput.value) { displayError("Please enter a username"); isValid = false; } if (!passwordInput.value) { displayError("Please enter a password"); isValid = false; } else if (passwordInput.value.length < 6) { displayError("Password should be at least 6 characters long"); isValid = false; } if (passwordInput.value !== confirmPasswordInput.value) { displayError("Passwords do not match"); isValid = false; } return isValid; } function displayError(message) { errorMessage.textContent = message; errorMessage.style.display = "block"; } });
